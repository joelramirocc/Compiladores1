Grammar
=======

input -> stmt_list


    stmt_list -> stmt_list ; stmt ;
                | stmt

stmt_list -> stmt stmt_listP
stmt_listP -> ; stmt ; stmt_listP
        | %eps


stmt -> IDENT assign
        | PRINT expr
        | fun_decl


assign -> = expr


fun_decl -> IDENT ( arg ) = expr


arg -> NUMBER
      | IDENT


    expr ->   expr + term
            | term

expr -> term exprP
exprP -> + term exprP
        | %eps


    term -> term * factor
            | factor

term -> factor termP
termP -> * factor termP
        | %eps

factor -> IDENT
        | NUMBER
        | ( expr )
